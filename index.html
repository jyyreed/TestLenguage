function doGet() { // Accede a tu hoja de cálculo de Google var ss = SpreadsheetApp.openById('1yfF8PUt0NmRZR-Ac7Eqk9FF9yUSeqx7z1PA5CIjRTbI'); var sheet = ss.getSheetByName('Idiomas'); // Obtén todas las filas desde la fila 3 hacia abajo, y las tres columnas var data = sheet.getRange(3, 2, sheet.getLastRow() - 2, 3).getValues(); var result = ""; // Recorre los datos y arma un string con las palabras de cada columna data.forEach(function(row) { var english = row[0]; // Palabra en inglés var spanish = row[1]; // Palabra en español var chinese = row[2]; // Palabra en chino result += english + '\n' + spanish + '\n' + chinese + '\n'; }); // Establecer los encabezados CORS para permitir solicitudes var output = ContentService.createTextOutput(result); output.setMimeType(ContentService.MimeType.TEXT); output.appendHeader('Access-Control-Allow-Origin', '*'); // Permitir todos los orígenes output.appendHeader('Access-Control-Allow-Methods', 'GET'); // Solo métodos GET return output; }
